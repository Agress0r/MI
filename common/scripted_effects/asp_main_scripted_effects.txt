# Директория для скриптовых событий
enough_minerals_in_system_algorithm_check = {
	owner = {
		set_variable = {
			which = minerals_in_system
			value = $SYS_MIN$
		}
	}
	if = {
		limit = {
			any_system_planet = {
			has_resource = {
				type = minerals
				amount = 1
			}
		}
	  }
	  change_variable = {
		which = minerals_in_system
		VALUE = @\[($SYS_MIN$ + 1)]
	}
    }
}
