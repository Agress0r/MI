# Директория для скриптовых событий

asp_system_mineral_check = {
	every_system_planet = {
			if = {
				limit = {
						num_minerals > 20
			}
				add_modifier = { modifier = habs_mineral_output_4 }
		}
		else_if = {
			limit = {
				AND = {
					num_minerals > 12
					any_owned_planet = {
					is_planet_class = pc_habitat
				}
			}
		}
			add_modifier = { modifier = habs_mineral_output_3 }
	}

	else_if = {
		limit = {
			AND = {
				num_minerals > 8
				any_owned_planet = {
				is_planet_class = pc_habitat
			}
		}
	}
		add_modifier = { modifier = habs_mineral_output_2 }
}
else = {
	add_modifier = { modifier = habs_mineral_output_1 }
}
  }
}
